## API

```tsx
const { isSupported, sRGBHex, open } = useEyeDropper(options)
```

### 选项 Options \{#options}

```tsx
export type UseEyeDropperOptions = {
  /**
   * 初始 sRGBHex。
   *
   * @defaultValue ''
   */
  initialValue?: string
  /**
   * 是否将 sRGBHex 转换为大写。
   *
   * @defaultValue true
   */
  upperCase?: boolean
  /**
   * 是否保留 sRGBHex 中的前导井号。
   *
   * @defaultValue true
   */
  keepLeadingHash?: boolean
}
```

### 返回值 \{#returns}

```tsx
export type UseEyeDropperReturns = {
  /**
   * 是否支持 EyeDropper。
   */
  isSupported: boolean
  /**
   * sRGBHex 颜色值。
   */
  sRGBHex: string
  /**
   * 打开 EyeDropper。
   *
   * @param options - 打开选项。
   */
  open: (options?: EyeDropperOpenOptions) => Promise<string | null>
}
```
