## API \{#api}

```tsx
const { text, rects, ranges, selectionRef } = useTextSelection()
```

### 返回值 \{#returns}

```tsx
export interface UseTextSelectionReturns {
  /**
   * 一个 React Ref 对象，持有当前的 Selection 对象。
   */
  selectionRef: MutableRefObject<Selection | null>
  /**
   * 所选的文本。
   */
  text: string
  /**
   * 每个选定范围的 DOMRect 对象。
   */
  rects: DOMRect[]
  /**
   * 每个选定范围的 Range 对象。
   */
  ranges: Range[]
}
```
