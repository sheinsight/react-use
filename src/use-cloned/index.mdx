## API

```tsx
const [cloned, setCloned, sync] = useCloned(state, options)
```

### State

Any valid JavaScript type, such as `string`, `number`, `object`, `array`, etc.

### Options

```tsx
export type UseClonedOptions = {
  /**
   * Clone function
   *
   * @defaultValue defaultCloneFn
   */
  clone?: <T>(source: T) => T
  /**
   * Whether to manually trigger the clone function
   *
   * @defaultValue false
   */
  manual?: boolean
  /**
   * Whether to deep compare the source object
   *
   * @defaultValue true
   */
  deep?: boolean
}
```

### Returns

```tsx
export type UseClonedReturns<T> = [
  /**
   * Cloned state
   */
  cloned: T,
  /**
   * Set cloned state
   */
  setCloned: ReactSetState<T>,
  /**
   * Sync cloned state with source state
   */
  sync: () => void,
]
```
