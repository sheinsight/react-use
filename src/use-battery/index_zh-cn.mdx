## API

```tsx
const battery = useBattery()
```

### 返回值 Returns \{#returns}

```tsx
export interface UseBatteryReturns {
  /**
   * 是否正在充电
   */
  charging: boolean
  /**
   * 电池电量，0 到 1 之间的浮点数
   */
  level: number
  /**
   * 完全充电所剩余的时间
   */
  chargingTime: number
  /**
   * 完全放电所剩余的时间
   */
  dischargingTime: number
  /**
   * 是否支持电池 API
   */
  isSupported: boolean
  /**
   * 更新电池状态
   */
  update(): void
}
```
