## API

```tsx
const battery = useBattery()
```

### Returns

```tsx
export type UseBatteryReturn = {
  /**
   * Whether the battery is charging
   */
  charging: boolean
  /**
   * The battery level as a floating point number between 0 and 1
   */
  level: number
  /**
   * The seconds remaining to charge the battery fully
   */
  chargingTime: number
  /**
   * The seconds remaining to discharge the battery fully
   */
  dischargingTime: number
  /**
   * Whether the battery API is supported
   */
  isSupported: boolean
  /**
   * Update the battery state
   */
  update(): void
}
```
