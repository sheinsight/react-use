### 选项 Options \{#options}

```tsx
export type UseFullscreenOptions = {
  /**
   * 组件被卸载时自动退出全屏模式
   *
   * @defaultValue false
   */
  autoExit?: boolean
}
```

### 返回值 Returns \{#returns}

```tsx
export interface UseFullscreenReturns {
  /**
   * 浏览器是否支持全屏 API
   */
  isSupported: boolean
  /**
   * 元素是否处于全屏模式
   */
  isFullscreen: boolean
  /**
   * 元素本身是否处于全屏模式
   */
  isSelfFullscreen: boolean
  /**
   * 进入全屏模式
   */
  enter(): Promise<void>
  /**
   * 退出全屏模式
   */
  exit(): Promise<void>
  /**
   * 切换全屏模式
   */
  toggle(): Promise<void>
}
```
