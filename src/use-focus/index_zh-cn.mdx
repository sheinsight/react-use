## API

```tsx
const focus = useFocus(elementTarget, options)
```

import { Tip } from '@/components'

### Options

```tsx
export type UseFocusOptions = {
  /**
   * initial focus state
   *
   * @defaultValue false
   */
  initialValue?: boolean
}
```

### Returns

```tsx
export type UseFocusReturns = readonly [
  /**
   * focus state
   */
  focused: boolean,
  /**
   * focus actions
   */
  actions: {
    /**
     * focus the target element
     */
    focus(): void
    /**
     * blur the target element
     */
    blur(): void
  },
]
```
